    <script>
        let currentSlide = 0;
        const totalSlides = 3;

        function nextSlide() {
            currentSlide = (currentSlide + 1) % totalSlides;
            updateSlider();
        }

        function prevSlide() {
            currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;
            updateSlider();
        }

        function updateSlider() {
            const cardSlider = document.getElementById('cardSlider');
            cardSlider.style.transform = 'translateX(-' + (currentSlide * 300) + 'px)';

            for (let i = 0; i < totalSlides; i++) {
                const bg = document.getElementById('sliderBg' + (i + 1));
                const text = document.getElementById('textSlide' + (i + 1));
                
                if (i === currentSlide) {
                    bg.style.opacity = '1';
                    text.style.opacity = '1';
                } else {
                    bg.style.opacity = '0';
                    text.style.opacity = '0';
                }
            }
        }
    </script>
